import pygame import sys # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è Pygame pygame.init() # –ü–∞—Ä–∞–º–µ—Ç—Ä—ã CELL_SIZE = 40 GRID_SIZE = 15 WIDTH = GRID_SIZE * CELL_SIZE HEIGHT = GRID_SIZE * CELL_SIZE + 80 # + –º–µ—Å—Ç–æ –ø–æ–¥ –∑–∞–≥–æ–ª–æ–≤–æ–∫ FONT_SIZE = 24 TITLE_FONT_SIZE = 32 # –¶–≤–µ—Ç–∞ BACKGROUND = (250, 250, 255) GRID_COLOR = (50, 50, 100) TEXT_COLOR = (0, 0, 0) HIGHLIGHT_COLOR = (255, 200, 0) WORD_LIST_BG = (240, 240, 255) # –°–æ–∑–¥–∞–Ω–∏–µ –æ–∫–Ω–∞ screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("–ù–æ–≤–æ–≥–æ–¥–Ω–∏–π –∫—Ä–æ—Å—Å–≤–æ—Ä–¥ üéÑ") # –®—Ä–∏—Ñ—Ç—ã font = pygame.font.SysFont('Arial', FONT_SIZE, bold=True) title_font = pygame.font.SysFont('Arial', TITLE_FONT_SIZE, bold=True) # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è –ø—É—Å—Ç–æ–π —Å–µ—Ç–∫–∏ grid = [['' for _ in range(GRID_SIZE)] for _ in range(GRID_SIZE)] # –°–ª–æ–≤–∞ –∏ –∏—Ö –ø–æ–∑–∏—Ü–∏–∏ [—Å–ª–æ–≤–æ, x, y, –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–µ ('h' –∏–ª–∏ 'v')] words = [ ("—ë–ª–∫–∞", 0, 1, 'h'), ("—Å–Ω–µ–≥—É—Ä–æ—á–∫–∞", 0, 3, 'h'), ("—Å–Ω–µ–≥", 0, 5, 'h'), ("–∫–∞—Ç–æ–∫", 0, 7, 'h'), ("–ø–æ–¥–∞—Ä–∫–∏", 2, 9, 'h'), ("—à–∞–º–ø–∞–Ω—Å–∫–æ–µ", 1, 11, 'h'), ("–≥–∏—Ä–ª—è–Ω–¥–∞", 6, 0, 'v'), ("—Å–Ω–µ–≥–æ–≤–∏–∫", 3, 0, 'v'), ("–º–∏—à—É—Ä–∞", 9, 2, 'v'), ("–ø—Ä–∞–∑–¥–Ω–∏–∫", 12, 3, 'v') ] # –ó–∞–ø–æ–ª–Ω–µ–Ω–∏–µ —Å–µ—Ç–∫–∏ for word, x, y, direction in words: for i, char in enumerate(word): if direction == 'h': grid[y][x + i] = char.upper() else: # 'v' grid[y + i][x] = char.upper() # –§—É–Ω–∫—Ü–∏—è –æ—Ç—Ä–∏—Å–æ–≤–∫–∏ —Å–µ—Ç–∫–∏ def draw_grid(): for row in range(GRID_SIZE): for col in range(GRID_SIZE): rect = pygame.Rect(col * CELL_SIZE, row * CELL_SIZE + 60, CELL_SIZE, CELL_SIZE) pygame.draw.rect(screen, GRID_COLOR, rect, 1) # —Ç–æ–Ω–∫–∞—è —Ä–∞–º–∫–∞ if grid[row][col]: text = font.render(grid[row][col], True, TEXT_COLOR) screen.blit(text, (rect.centerx - text.get_width()//2, rect.centery - text.get_height()//2)) # –§—É–Ω–∫—Ü–∏—è –æ—Ç—Ä–∏—Å–æ–≤–∫–∏ —Å–ø–∏—Å–∫–∞ —Å–ª–æ–≤ def draw_word_list(): pygame.draw.rect(screen, WORD_LIST_BG, (0, HEIGHT - 80, WIDTH, 80)) y_offset = HEIGHT - 65 cols = 5 for i, (word, _, _, _) in enumerate(words): text = font.render(f"{i+1}. {word}", True, TEXT_COLOR) x = (WIDTH // cols) * (i % cols) + 20 y = y_offset + (i // cols) * 25 screen.blit(text, (x, y)) # –ì–ª–∞–≤–Ω—ã–π —Ü–∏–∫–ª running = True while running: screen.fill(BACKGROUND) # –ó–∞–≥–æ–ª–æ–≤–æ–∫ title = title_font.render("–ù–û–í–û–ì–û–î–ù–ò–ô –ö–†–û–°–°–í–û–†–î üéÖ", True, (30, 30, 80)) screen.blit(title, (WIDTH//2 - title.get_width()//2, 10)) draw_grid() draw_word_list() for event in pygame.event.get(): if event.type == pygame.QUIT: running = False elif event.type == pygame.KEYDOWN: if event.key == pygame.K_ESCAPE: running = False pygame.display.flip() pygame.quit() sys.exit()
